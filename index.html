function shuffle(array) { for (let i = array.length - 1; i > 0; i--) { const j = Math.floor(Math.random() * (i + 1)); [array[i], array[j]] = [array[j], array[i]]; } return array; } function generateMatching() { // 1. 입력값 가져오기 let maleNames = document.getElementById('maleInput').value.split(',').map(name => name.trim()).filter(name => name !== ""); let femaleNames = document.getElementById('femaleInput').value.split(',').map(name => name.trim()).filter(name => name !== ""); // 2. 고정 매칭 정의 const fixedPairs = [ { male: "봉쥬", female: "양뽁" }, { male: "찐빵", female: "만두" } ]; // 3. 인원 수 체크 (예: 13명 기준) if (maleNames.length < 2 || femaleNames.length < 2) { alert("최소한 고정 매칭 인원 이상의 명단이 필요합니다."); return; } // 4. 고정 매칭 대상자를 명단에서 제외 (나머지 인원만 섞기 위함) fixedPairs.forEach(pair => { maleNames = maleNames.filter(name => name !== pair.male); femaleNames = femaleNames.filter(name => name !== pair.female); }); // 5. 남은 인원들만 랜덤하게 섞기 const shuffledMales = shuffle([...maleNames]); const shuffledFemales = shuffle([...femaleNames]); // 6. 결과 리스트 생성 (고정 커플 먼저 넣기) const finalMatching = []; // 고정 커플 추가 fixedPairs.forEach(pair => { finalMatching.push({ male: pair.male, female: pair.female }); }); // 나머지 섞인 인원 추가 (남은 인원 수만큼 반복) const remainingCount = Math.min(shuffledMales.length, shuffledFemales.length); for (let i = 0; i < remainingCount; i++) { finalMatching.push({ male: shuffledMales[i], female: shuffledFemales[i] }); } // 7. 결과 출력 (전체 리스트를 한 번 더 섞어서 고정 커플이 항상 1, 2번에 나오지 않게 함) shuffle(finalMatching); const resultBody = document.getElementById('resultBody'); resultBody.innerHTML = ""; finalMatching.forEach((pair, index) => { const row = ` 자리 ${index + 1} ${pair.male} ${pair.female} `; resultBody.innerHTML += row; }); document.getElementById('resultTable').style.display = 'table'; }
